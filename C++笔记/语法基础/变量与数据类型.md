## 类型转换
### 隐式转换
```Cpp
#include <iostream>
using namespace std;

int main() {
    int intValue = 42;
    double doubleValue = intValue;    // int 自动转换为 double
    
    cout << "整数值：" << intValue << endl;
    cout << "转换后的浮点值：" << doubleValue << endl;
    
    // 运算中的自动类型转换
    int a = 10;
    double b = 3.14;
    double result = a + b;  // a 自动转换为 double
    
    cout << "10 + 3.14 = " << result << endl;
    
    return 0;
}

```
### 显式转换
```cpp
#include <iostream>
using namespace std;

int main() {
    double pi = 3.14159;
    
    // C 风格的强制转换
    int intPi1 = (int)pi;
    
    // C++ 风格的强制转换
    int intPi2 = static_cast<int>(pi);
    
    cout << "原始值：" << pi << endl;
    cout << "C 风格转换：" << intPi1 << endl;
    cout << "C++ 风格转换：" << intPi2 << endl;
    
    // 字符和数字的转换
    char digit = '5';
    int number = digit - '0';  // 字符转数字的技巧
    
    cout << "字符 '" << digit << "' 转换为数字：" << number << endl;
    
    return 0;
}
```
